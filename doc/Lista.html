def insertar(x)
    nodo = Nodo.new(x, nil, nil)   
    if @head == nil && @tail==nil
        nodo.next=@head
        nodo.prev=@tail
        @head=nodo
        @tail=nodo
    else
        nodo.prev=@head
        nodo.next=nil
        @head.next=nodo
        @head=nodo
    end
    
end
