def nplatos()#Obtenemos el conjunto de platos (con sus descripciones, porciones y gramos)
            plat=''
            i=0
            while i < @descripcion.size
                    plat=plat + "- #{@descripcion[i]}, #{@porcion[i]}, #{@gramos[i]}\n"
                    i=i+1
            end
            return plat
        end
